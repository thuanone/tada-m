
# Retrieve a Knative Service

Verb: GET
URL-Template: https://<coligo-serving-kube-api-proxy-domain>/apis/serving.knative.dev/v1alpha1/namespaces/<k8s-namespace-name>/services/<kn-service-name>
Headers:
 - 'Accept': 'application/json'
 - 'Authorization': `Bearer ${accessToken}`
 - 'Content-Type': 'application/json'
 - 'cache-control': 'no-cache'
